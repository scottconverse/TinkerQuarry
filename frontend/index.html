include <BOSL2/std.scad>
include <BOSL2/screws.scad>

// Pi 4 wall bracket — M3, 3mm walls
wall = 3;  screw = "M3";

module bracket() {
  diff() {
    cuboid([120,80,wall], rounding=2, edges="Z");
    // mounting holes — top face
    move_copies(grid(2, spacing=58))
      screw_hole(screw, anchor=TOP);
  }
}
bracket();
